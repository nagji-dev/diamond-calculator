function calculateWeight() { let shape = document.getElementById("shape").value; let L = parseFloat(document.getElementById("length").value); let W = parseFloat(document.getElementById("width").value); let depthPercent = parseFloat(document.getElementById("depth").value); let weight = 0; let depthMM = 0; // ROUND (NO CHANGE) if(shape === "round"){ depthMM = (L * depthPercent) / 100; weight = L * L * depthMM * 0.0061; } // OVAL else if(shape === "oval"){ depthMM = (L * depthPercent) / 100; weight = L * W * depthMM * 0.0068; } // PEAR else if(shape === "pear"){ depthMM = (L * depthPercent) / 100; weight = L * W * depthMM * 0.0069; } // MARQUISE else if(shape === "marquise"){ depthMM = (L * depthPercent) / 100; weight = L * W * depthMM * 0.0059; } // CUSHION else if(shape === "cushion"){ depthMM = (L * depthPercent) / 100; weight = L * W * depthMM * 0.0067; } // EMERALD else if(shape === "emerald"){ depthMM = (L * depthPercent) / 100; weight = L * W * depthMM * 0.0065; } // Ratio let ratio = (L / W).toFixed(2); // RESULT document.getElementById("weightResult").innerHTML = "Weight : " + weight.toFixed(2) + " ct"; document.getElementById("ratioResult").innerHTML = "Ratio : " + ratio; }
